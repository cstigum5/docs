---
title: "List Jobs"
description: "Administrators, impersonating a service user, can request a list of jobs for a child account."
api: GET https://cloud.cdata.com/api/job/list
contentType: application/json
---


## Responses

### 200

`HTTP 200 OK`

<Expandable title="Response body">

<ResponseField name="jobs" type="array of objects" required>
  <ResponseField name="jobType" type="string" required>
    The type of job the user wants to run.
  </ResponseField>
  <ResponseField name="id" type="string" required>
    The job's Id.
  </ResponseField>
  <ResponseField name="name" type="string" required>
    The job's name.
  </ResponseField>
  <ResponseField name="enabled" type="boolean" required>
    A Boolean value indicating whether the job is enabled.
  </ResponseField>
  <ResponseField name="created" type="string" required>
    The date and time the job was created.
  </ResponseField>
  <ResponseField name="lastModified" type="string" required>
    The date and time the job was last modified.
  </ResponseField>
  <ResponseField name="nextRunTime" type="string" required>
    The date and time the job is scheduled to be run next.
  </ResponseField>
  <ResponseField name="intervalUnit" type="string" required>
    A string representing the interval unit for the job (Hour/Day/Week/Month).
  </ResponseField>
  <ResponseField name="intervalValue" type="integer" required>
    An integer that represents how many of the selected interval unit the job should wait before running again.
  </ResponseField>
  <ResponseField name="logVerbosity" type="integer" required>
    A number indicating the level of log verbosity (1-5).
  </ResponseField>
  <ResponseField name="lastRun" type="object">
    The last run metrics for the job, including the date and time, duration, status, rows affected, and next run time.
  </ResponseField>
  <ResponseField name="params" type="object">
    An object containing different job details, depending on whether the job is a cache job or a scheduled query.
  </ResponseField>
</ResponseField>

</Expandable>
